import { useChatStore } from '@/store';
import { useState } from 'react';
import { memAPI } from '@/api';

export function useChat() {
  const store = useChatStore();
  const [streaming, setStreaming] = useState(false);

  const sendStreamMessage = async (message: string) => {
    setStreaming(true);
    store.addMessage({ role: 'user', content: message, timestamp: new Date().toISOString() });
    
    let assistantMessage = '';
    
    await memAPI.chatStream(
      message,
      (chunk) => {
        assistantMessage += chunk;
        store.updateLastAssistantMessage(assistantMessage);
      },
      (error) => {
        console.error('Stream error:', error);
        setStreaming(false);
      },
      () => {
        setStreaming(false);
      }
    );
  };

  return {
    messages: store.messages,
    loading: store.loading || streaming,
    sendMessage: sendStreamMessage,
    clearMessages: store.clearMessages,
  };
}
